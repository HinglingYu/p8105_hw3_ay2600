``` r
# How many aisles
num_aisles = 
  instacart %>% 
  group_by(aisle)%>% 
  summarize(n_obs = n())

# Find out the aisles with most items ordered
most_ordered_aisle =
  instacart %>% 
  group_by(aisle) %>% 
  summarize(total_items_ordered = n()) %>% 
  arrange(-total_items_ordered)
```
